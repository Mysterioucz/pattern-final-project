```python
"""
Train GCN with skip connections on the Elliptic Bitcoin dataset using the
selected class-imbalance undersampling method.

Per D-02: Uses GCN(use_skip=True) from projects/src/models/gcn.py.
Per D-03: Sampling is applied per-timestep to training snapshots only.
Per D-05: load_snapshots(filter_unknown=False) keeps unknown nodes in graph.
Per D-06: --sampler flag selects the method; 'enn' is the default.
Per D-08: Primary metric is Illicit F1 (pos_label=0).
Per D-09: Results appended to projects/results/sampling_comparison.csv.

Usage:
    # From repo root:
    python projects/src/train_gcn_skip.py                           # ENN (default)
    python projects/src/train_gcn_skip.py --sampler nearmiss1
    python projects/src/train_gcn_skip.py --sampler none --epochs 2  # smoke test

Reference hyperparameters (from reference/EvolveGCNFai/train_gcn_skip.py):
    NUM_EPOCH=500, LEARNING_RATE=1e-3, hidden_channels=[64]
"""

from __future__ import annotations

import argparse
import csv
import os
import sys
from pathlib import Path

import torch
import torch.nn.functional as F
from sklearn.metrics import f1_score, precision_score, recall_score
from torch_geometric.data import Data
from torch_geometric.utils import to_undirected

# ── sys.path: allow "python projects/src/train_gcn_skip.py" from repo root ──
_SRC_DIR = Path(__file__).resolve().parent
if str(_SRC_DIR) not in sys.path:
    sys.path.insert(0, str(_SRC_DIR))

from data.preprocessor import load_snapshots
from data.sampler import apply_sampling
from models.gcn import GCN

# ── Results output path (D-09) ───────────────────────────────────────────────
_REPO_ROOT = _SRC_DIR.parent.parent  # projects/src/ -> projects/ -> repo root
RESULTS_CSV = _REPO_ROOT / "projects" / "results" / "sampling_comparison.csv"


# ---------------------------------------------------------------------------
# Training loop (D-03: sampling hook applied per-timestep, train only)
# ---------------------------------------------------------------------------

def train(
    model: GCN,
    train_snapshots: list[dict],
    optimizer: torch.optim.Optimizer,
    sampler_name: str,
    epochs: int = 500,
) -> None:
    """Train the GCN-skip model over all training timestep snapshots.

    Sampling is applied only to training snapshots (D-03, D-04).
    Unknown nodes are excluded from the loss (y != 2 mask per D-05).
    """
    for epoch in range(1, epochs + 1):
        model.train()
        epoch_loss = 0.0
        for snap in train_snapshots:
            # Apply to_undirected BEFORE Data construction (loader.py convention)
            data = Data(
                x=snap["x"],
                edge_index=to_undirected(snap["edge_index"]),
                y=snap["y"],
            )
            # Apply sampler to training snapshot only (Pitfall 4 — never sample test)
            if sampler_name != "none":
                data = apply_sampling(data, sampler_name)   # D-03

            optimizer.zero_grad()
            out = model(data, return_logits=True)            # (N, 2) logits
            labeled_mask = data.y != 2                       # exclude unknowns (D-05)
            loss = F.cross_entropy(out[labeled_mask], data.y[labeled_mask])
            loss.backward()
            optimizer.step()
            epoch_loss += loss.item()

        if epoch % 50 == 0:
            print(f"Epoch {epoch:04d} | Loss: {epoch_loss:.4f}")


# ---------------------------------------------------------------------------
# Evaluation (D-08: Illicit F1 as primary metric)
# ---------------------------------------------------------------------------

def evaluate(
    model: GCN,
    test_snapshots: list[dict],
) -> tuple[float, float, float]:
    """Evaluate across all test snapshots.

    Test snapshots are NEVER sampled — evaluated on original unmodified graphs.

    Returns:
        (illicit_f1, precision, recall) — all computed with pos_label=0 (D-08).
    """
    model.eval()
    y_true_all: list[int] = []
    y_pred_all: list[int] = []

    with torch.no_grad():
        for snap in test_snapshots:
            data = Data(
                x=snap["x"],
                edge_index=to_undirected(snap["edge_index"]),
                y=snap["y"],
            )
            out = model(data, return_logits=True)
            pred = out.argmax(dim=1)
            labeled_mask = data.y != 2
            y_true_all.extend(data.y[labeled_mask].cpu().tolist())
            y_pred_all.extend(pred[labeled_mask].cpu().tolist())

    # D-08: pos_label=0 because illicit is encoded as 0
    illicit_f1 = f1_score(y_true_all, y_pred_all, pos_label=0, average="binary", zero_division=0)
    precision  = precision_score(y_true_all, y_pred_all, pos_label=0, average="binary", zero_division=0)
    recall     = recall_score(y_true_all, y_pred_all, pos_label=0, average="binary", zero_division=0)

    return illicit_f1, precision, recall


# ---------------------------------------------------------------------------
# CSV output (D-09)
# ---------------------------------------------------------------------------

def append_results(method: str, illicit_f1: float, precision: float, recall: float) -> None:
    """Append one row to projects/results/sampling_comparison.csv (D-09).

    Creates the file and header on first write (Pitfall 8: makedirs guard).
    """
    os.makedirs(RESULTS_CSV.parent, exist_ok=True)
    write_header = not RESULTS_CSV.exists()
    with open(RESULTS_CSV, "a", newline="") as f:
        writer = csv.DictWriter(
            f,
            fieldnames=["method", "test_illicit_f1", "test_precision", "test_recall"],
        )
        if write_header:
            writer.writeheader()
        writer.writerow({
            "method": method,
            "test_illicit_f1": f"{illicit_f1:.4f}",
            "test_precision":  f"{precision:.4f}",
            "test_recall":     f"{recall:.4f}",
        })
    print(f"\nResults appended to: {RESULTS_CSV}")


def print_summary_table(method: str, illicit_f1: float, precision: float, recall: float) -> None:
    """Print a human-readable comparison table to stdout (D-09)."""
    print("\n" + "=" * 50)
    print(f"  GCN-Skip Sampling Ablation — Method: {method}")
    print("=" * 50)
    print(f"  {'Metric':<22} {'Value':>8}")
    print(f"  {'-'*22} {'-'*8}")
    print(f"  {'Illicit F1 (primary)':22} {illicit_f1:>8.4f}")
    print(f"  {'Illicit Precision':22} {precision:>8.4f}")
    print(f"  {'Illicit Recall':22} {recall:>8.4f}")
    print("=" * 50)
    print("  Reference TF baseline (ENN, filter_unknown=False): F1 ~0.364")
    print("=" * 50 + "\n")


# ---------------------------------------------------------------------------
# Entry point
# ---------------------------------------------------------------------------

def main() -> None:
    parser = argparse.ArgumentParser(
        description="GCN-Skip training with class-imbalance sampling ablation"
    )
    parser.add_argument(
        "--sampler",
        type=str,
        default="enn",
        choices=["enn", "edge_pruning", "cluster_centroid", "nearmiss1", "nearmiss2", "none"],
        help="Sampling method to apply to training snapshots (default: enn)",
    )
    parser.add_argument(
        "--epochs",
        type=int,
        default=500,
        help="Number of training epochs (default: 500, reference: 500)",
    )
    parser.add_argument(
        "--lr",
        type=float,
        default=1e-3,
        help="Adam learning rate (default: 1e-3, reference: 1e-3)",
    )
    args = parser.parse_args()

    print(f"\n[GCN-Skip] Sampler: {args.sampler} | Epochs: {args.epochs} | LR: {args.lr}")

    # Load snapshots — filter_unknown=False per D-05 (unknowns stay in graph)
    print("[GCN-Skip] Loading snapshots (filter_unknown=False)...")
    snapshots = load_snapshots(normalize=True, filter_unknown=False)
    train_snapshots = [s for s in snapshots if s["split"] == "train"]
    test_snapshots  = [s for s in snapshots if s["split"] == "test"]
    print(f"[GCN-Skip] Train snapshots: {len(train_snapshots)} | Test snapshots: {len(test_snapshots)}")

    # Derive num_node_features at runtime (Pitfall 6: 165, not 166)
    num_node_features = snapshots[0]["x"].shape[1]
    print(f"[GCN-Skip] num_node_features = {num_node_features}")

    # Initialize GCN with skip connection (D-02)
    model = GCN(
        num_node_features=num_node_features,
        hidden_channels=[64],
        use_skip=True,        # D-02: skip branch = nn.Linear(num_node_features, 2)
    )
    optimizer = torch.optim.Adam(model.parameters(), lr=args.lr)

    print(f"[GCN-Skip] Training for {args.epochs} epochs...")
    train(model, train_snapshots, optimizer, sampler_name=args.sampler, epochs=args.epochs)

    print("[GCN-Skip] Evaluating on test snapshots (unsampled)...")
    illicit_f1, precision, recall = evaluate(model, test_snapshots)

    print_summary_table(args.sampler, illicit_f1, precision, recall)
    append_results(args.sampler, illicit_f1, precision, recall)


if __name__ == "__main__":
    main()
```
