"""
Undersampling module for GCN-Skip training on the Elliptic Bitcoin dataset.

Implements five class-imbalance sampling methods applied per-timestep to labeled
nodes only. Unknown nodes (y == 2) always pass through untouched (D-05).

Supported methods (D-07):
  'enn'              — EditedNearestNeighbours (default, D-06)
  'edge_pruning'     — Custom ENN on graph neighbourhood (worse than ENN)
  'cluster_centroid' — ClusterCentroids with voting='hard' (worse than ENN)
  'nearmiss1'        — NearMiss version=1 (worse than ENN)
  'nearmiss2'        — NearMiss version=2 (worse than ENN)

Usage:
    from data.sampler import apply_sampling, get_sampler
    data_sampled = apply_sampling(data, 'enn')
"""

from __future__ import annotations

import numpy as np
import torch
from torch_geometric.data import Data
from torch_geometric.utils import subgraph
from imblearn.under_sampling import (
    ClusterCentroids,
    EditedNearestNeighbours,
    NearMiss,
)


# ---------------------------------------------------------------------------
# Public factory (D-07)
# ---------------------------------------------------------------------------

def get_sampler(name: str):
    """Return an initialised imblearn sampler for the given method name.

    Args:
        name: One of 'enn', 'edge_pruning', 'cluster_centroid',
              'nearmiss1', 'nearmiss2'. Case-insensitive.

    Returns:
        An imblearn BaseUnderSampler instance, or None for 'edge_pruning'
        (which uses a custom code path in apply_sampling).

    Raises:
        ValueError: If name is not a recognised sampler name.
    """
    name = name.lower().strip()
    if name == "enn":
        return EditedNearestNeighbours(sampling_strategy="auto", n_neighbors=3)
    elif name == "nearmiss1":
        return NearMiss(version=1, n_neighbors=3)
    elif name == "nearmiss2":
        return NearMiss(version=2, n_neighbors=3)
    elif name == "cluster_centroid":
        return ClusterCentroids(voting="hard", random_state=42)
    elif name == "edge_pruning":
        return None  # sentinel: handled by custom _edge_pruning() in apply_sampling
    else:
        raise ValueError(
            f"Unknown sampler: {name!r}. "
            "Choose from: enn, edge_pruning, cluster_centroid, nearmiss1, nearmiss2"
        )


# ---------------------------------------------------------------------------
# Public transform (D-03, D-04, D-05)
# ---------------------------------------------------------------------------

def apply_sampling(data: Data, sampler_name: str) -> Data:
    """Apply the named sampling method to a single-timestep PyG Data object.

    Per D-03: sampling is applied in node feature space, per-timestep.
    Per D-04: incident edges of removed nodes are removed (induced subgraph).
    Per D-05: unknown nodes (y == 2) always survive; sampler operates only
              on labeled nodes (illicit=0, licit=1).

    Args:
        data:         PyG Data with x (N, F), edge_index (2, E), y (N,).
        sampler_name: Method name — see get_sampler() for supported values.

    Returns:
        New Data object with surviving nodes and their induced edge subgraph.
    """
    y = data.y  # LongTensor (N,)

    # ── 1. Separate labeled vs unknown nodes ──────────────────────────────
    labeled_global_idx = (y != 2).nonzero(as_tuple=False).squeeze(1)   # LongTensor
    unknown_global_idx = (y == 2).nonzero(as_tuple=False).squeeze(1)   # LongTensor

    # Edge case: no labeled nodes — return data unchanged
    if labeled_global_idx.numel() == 0:
        return data

    X_labeled = data.x[labeled_global_idx].numpy()  # (N_labeled, F)
    y_labeled = y[labeled_global_idx].numpy()        # (N_labeled,)

    # ── 2. Apply sampling method ──────────────────────────────────────────
    sampler_name_lower = sampler_name.lower().strip()

    if sampler_name_lower == "edge_pruning":
        # Custom graph-neighbourhood ENN (D-06)
        surviving_global_list = _edge_pruning(data)
        surviving_global_labeled = torch.tensor(
            [i for i in surviving_global_list if y[i].item() != 2],
            dtype=torch.long,
        )
    elif sampler_name_lower == "cluster_centroid":
        # ClusterCentroids with voting='hard' — no sample_indices_; use row-match
        cc = ClusterCentroids(voting="hard", random_state=42)
        X_res, _ = cc.fit_resample(X_labeled, y_labeled)
        surviving_local = _recover_indices_by_row_match(X_labeled, X_res)
        surviving_global_labeled = labeled_global_idx[surviving_local]
    elif sampler_name_lower in ("nearmiss1", "nearmiss2"):
        version = 1 if sampler_name_lower == "nearmiss1" else 2
        # Guard: clamp n_neighbors to minority class count (Pitfall 2)
        n_illicit = int((y_labeled == 0).sum())
        n_neighbors = min(3, n_illicit) if n_illicit > 0 else 1
        nm = NearMiss(version=version, n_neighbors=n_neighbors)
        _, _ = nm.fit_resample(X_labeled, y_labeled)
        surviving_global_labeled = labeled_global_idx[nm.sample_indices_]
    else:
        # ENN (default) and any other imblearn-backed sampler with sample_indices_
        sampler = get_sampler(sampler_name_lower)
        # Guard: clamp n_neighbors for ENN as well (Assumption A2)
        n_min = int((y_labeled == 0).sum())
        if hasattr(sampler, "n_neighbors") and n_min < getattr(sampler, "n_neighbors", 3):
            sampler = EditedNearestNeighbours(
                sampling_strategy="auto",
                n_neighbors=max(1, n_min),
            )
        _, _ = sampler.fit_resample(X_labeled, y_labeled)
        surviving_global_labeled = labeled_global_idx[sampler.sample_indices_]

    # ── 3. Combine surviving labeled + all unknowns (D-05) ────────────────
    all_surviving = torch.cat([surviving_global_labeled, unknown_global_idx])
    all_surviving_sorted, _ = all_surviving.sort()

    # ── 4. Rebuild graph (D-04): induced subgraph via PyG utility ─────────
    new_edge_index, _ = subgraph(
        all_surviving_sorted,
        data.edge_index,
        relabel_nodes=True,
        num_nodes=data.num_nodes,
    )

    new_x = data.x[all_surviving_sorted]
    new_y = data.y[all_surviving_sorted]

    return Data(x=new_x, edge_index=new_edge_index, y=new_y)


# ---------------------------------------------------------------------------
# Private helpers
# ---------------------------------------------------------------------------

def _edge_pruning(data: Data) -> list[int]:
    """Return global node indices surviving edge-pruning (ENN on graph neighbourhood).

    For each majority-class (licit=1) node: if the majority of its labeled graph
    neighbours disagree with its label, the node is removed. Minority (illicit=0)
    and unknown (y=2) nodes are always kept (consistent with ENN's not-minority strategy).

    Args:
        data: PyG Data object (original snapshot, not the labeled-only subset).

    Returns:
        List of global node indices that survive.
    """
    y = data.y
    N = data.num_nodes
    src = data.edge_index[0].tolist()
    dst = data.edge_index[1].tolist()

    # Build undirected adjacency list
    adj: dict[int, set[int]] = {i: set() for i in range(N)}
    for s, d in zip(src, dst):
        adj[s].add(d)
        adj[d].add(s)

    surviving = []
    for n in range(N):
        label = y[n].item()
        if label != 1:
            # Unknown (2) and illicit (0): always keep (D-05)
            surviving.append(n)
        else:
            # Licit (majority): apply ENN-on-graph rule
            labeled_nbr_labels = [
                y[j].item() for j in adj[n] if y[j].item() != 2
            ]
            if len(labeled_nbr_labels) == 0:
                # No labeled neighbours — no information to remove on (Pitfall 7)
                surviving.append(n)
            elif sum(lbl != label for lbl in labeled_nbr_labels) > len(labeled_nbr_labels) / 2:
                pass  # Majority of neighbours disagree → remove
            else:
                surviving.append(n)
    return surviving


def _recover_indices_by_row_match(X_original: np.ndarray, X_resampled: np.ndarray) -> np.ndarray:
    """Recover local indices into X_original for each row in X_resampled.

    Used for ClusterCentroids(voting='hard') which does not expose sample_indices_.
    With voting='hard', each row in X_resampled is guaranteed to be an exact copy
    of the nearest real sample, so L1 distance to the closest row is ~0.

    Args:
        X_original:  (N, F) numpy array of labeled node features.
        X_resampled: (M, F) numpy array from cc.fit_resample(); M <= N.

    Returns:
        ndarray of shape (M,) with local indices into X_original.
    """
    surviving_local = []
    for row in X_resampled:
        diffs = np.abs(X_original - row).sum(axis=1)
        surviving_local.append(int(np.argmin(diffs)))
    return np.array(surviving_local, dtype=np.intp)
